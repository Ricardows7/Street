# Open the image file again
img = Image.open(img_path)

# Convert the image to RGBA (if not already in that mode)
img = img.convert("RGBA")

# Get data of the image
data = img.getdata()

# Create a new image data with transparency
new_data = []
for item in data:
    # Check if the pixel is black
    if item[:3] == (0, 0, 0):
        # Change the black to a new transparent value
        new_data.append((0, 0, 0, 0))
    else:
        new_data.append(item)

# Update image data
img.putdata(new_data)

# Save the new image with transparency
output_path_corrected = "/mnt/data/donatello_transparent_corrected.png"
img.save(output_path_corrected)
